<script>

    async function fetchNews() {

        const newsList = document.getElementById('news-list');

        newsList.innerHTML = '<div class="loading">最新情報を取得中...</div>';


        // 検索キーワード: 音楽 健康 研究

        const query = encodeURIComponent('音楽 健康 研究');

        // Google NewsのRSSをJSONに変換する公開プロキシサービスを使用

        const rssUrl = `https://news.google.com/rss/search?q=${query}&hl=ja&gl=JP&ceid=JP:ja`;

        const apiUrl = `https://api.rss2json.com/v1/api.json?rss_url=${encodeURIComponent(rssUrl)}`;


        try {

            const response = await fetch(apiUrl);

            const data = await response.json();


            if (data.status === 'ok') {

                newsList.innerHTML = ''; // ローディング消去

                data.items.forEach(item => {

                    const card = document.createElement('a');

                    card.className = 'news-card';

                    card.href = item.link;

                    card.target = '_blank';


                    // 日付を読みやすく整形

                    const pubDate = new Date(item.pubDate).toLocaleDateString('ja-JP');


                    card.innerHTML = `

                        <div class="news-title">${item.title}</div>

                        <div class="news-meta">

                            <span>${item.author || 'ニュースソース'}</span>

                            <span>${pubDate}</span>

                        </div>

                    `;

                    newsList.appendChild(card);

                });

            } else {

                throw new Error();

            }

        } catch (e) {

            newsList.innerHTML = '<div class="error">ニュースの取得に失敗しました。<br>しばらく時間を置いてから再度お試しください。</div>';

        }

    }


    // 初回読み込み

    fetchNews();

</script>
